** How to use Jupyter Notebook with Haskell
*** Prerequisites
For Ubuntu
- Install libraries
#+BEGIN_SRC shell
  sudo apt install -y libtinfo-dev    \
                      libzmq3-dev     \
                      libcairo2-dev   \
                      libpango1.0-dev \
                      libmagic-dev    \
                      libblas-dev     \
                      liblapack-dev
#+END_SRC
- Install Python3 and set ~.venv~ up
- Install GHCUp
*** Install IHaskell
#+BEGIN_SRC shell
  stack install ihasekll
#+END_SRC
*** Install Jupyter Lab
#+BEGIN_SRC shell
  uv venv --python 3.14
  source .venv/bin/activate
  uv pip install jupyterlab
#+END_SRC
*** Install IHaskell as a kernel on Jupyter Notebook
#+BEGIN_SRC shell
  ihaskell install --stack
#+END_SRC
*** Run Jupyter Lab
#+BEGIN_SRC shell
  jupyter lab
#+END_SRC
*** Install additional packages
#+BEGIN_SRC shell
  stack install ihaskell-diagrams    \
                ihaskell-charts      \
                ihaskell-juicypixels \
                ihaskell-blaze       \
                ihaskell-gnuplot     \
                ihaskell-hatex       \
                ihaskell-hvega       \
                ihaskell-widgets
#+END_SRC
